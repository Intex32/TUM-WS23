// ---------------------------------------------------------------
// example function for:
//      1          | if A and B are 1
//      1          | if X and Y are 0
//      don't care | otherwise
// ---------------------------------------------------------------
// F(A, B, X, Y) = (A * B) + (!X * !Y)
// ---------------------------------------------------------------
// put function for X in the free space down below
// ---------------------------------------------------------------
X(A, B, C, D)=(A*B*C*!D)+(!B*!C*!D)
// ---------------------------------------------------------------
// put function for Y in the free space down below
// ---------------------------------------------------------------
Y(A, B, C, D)=(A*B*!C*!D)+(!B*!C*!D)
// ---------------------------------------------------------------
// put function for Z in the free space down below
// ---------------------------------------------------------------
Z(A, B, C, D)=(!A*!B*!C*!D)
// ---------------------------------------------------------------
